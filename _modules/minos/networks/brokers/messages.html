        data: Any,
        service_name: str,

        status: Optional[BrokerMessageStatus] = None,

    ):
        if status is None:
            status = BrokerMessageStatus.SUCCESS
        if strategy is None:
            strategy = BrokerMessageStrategy.UNICAST
        super().__init__(topic, data, service_name, status=status, strategy=strategy, **kwargs)
